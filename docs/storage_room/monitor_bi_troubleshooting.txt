NextOccurence=
	FIRSTNONBLANK(
		SELECTCOLUMNS(
			FILTER(
				quipment_log,
				AND(
					EARLIER(equipment_log[equipment])=equipment_log[equipment],
					EARLIER(equipment_log[timestamp])<equipment_log[timestamp])
),
'timestamp',
[timestamp]
),1
)
